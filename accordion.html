import { useState } from 'react';
import { ChevronDown } from 'lucide-react';

const AccordionItem = ({ title, children, isOpen, onToggle }) => (
  <div className="border-b border-pearl-200 last:border-b-0">
    <button
      onClick={onToggle}
      className="w-full px-5 py-4 bg-white text-left text-base font-semibold text-pearl-900 flex justify-between items-center hover:bg-pearl-50 transition-colors"
    >
      <span>{title}</span>
      <ChevronDown
        className={`w-5 h-5 text-pearl-500 transition-transform ${
          isOpen ? 'rotate-180' : ''
        }`}
      />
    </button>
    <div
      className={`overflow-hidden transition-all duration-300 ${
        isOpen ? 'max-h-96' : 'max-h-0'
      }`}
    >
      <div className="px-5 pb-4 text-pearl-700 leading-relaxed">
        {children}
      </div>
    </div>
  </div>
);

const Accordion = ({ items, allowMultiple = false }) => {
  const [openItems, setOpenItems] = useState(new Set());

  const toggleItem = (index) => {
    setOpenItems((prev) => {
      const newSet = allowMultiple ? new Set(prev) : new Set();
      if (prev.has(index)) {
        newSet.delete(index);
      } else {
        newSet.add(index);
      }
      return newSet;
    });
  };

  return (
    <div className="border border-pearl-200 rounded-md overflow-hidden">
      {items.map((item, index) => (
        <AccordionItem
          key={index}
          title={item.title}
          isOpen={openItems.has(index)}
          onToggle={() => toggleItem(index)}
        >
          {item.content}
        </AccordionItem>
      ))}
    </div>
  );
};

// Product Accordion with Rich Content
const ProductAccordion = ({ specifications, certifications, shipping }) => {
  const [openSection, setOpenSection] = useState('specs');

  return (
    <div className="border border-pearl-200 rounded-md overflow-hidden">
      {/* Specifications */}
      <AccordionItem
        title="Specifications"
        isOpen={openSection === 'specs'}
        onToggle={() => setOpenSection(openSection === 'specs' ? '' : 'specs')}
      >
        <table className="w-full text-sm">
          {specifications.map((spec, i) => (
            <tr key={i} className="border-b border-pearl-100 last:border-0">
              <td className="py-2 text-pearl-600 font-medium">{spec.label}</td>
              <td className="py-2 text-pearl-900">{spec.value}</td>
            </tr>
          ))}
        </table>
      </AccordionItem>

      {/* Certifications */}
      <AccordionItem
        title="Certifications"
        isOpen={openSection === 'certs'}
        onToggle={() => setOpenSection(openSection === 'certs' ? '' : 'certs')}
      >
        <ul className="space-y-2">
          {certifications.map((cert, i) => (
            <li key={i} className="flex items-center gap-2">
              <span className="text-coral-500">✓</span>
              {cert}
            </li>
          ))}
        </ul>
      </AccordionItem>
    </div>
  );
};

Usage Guidelines

✅ Do

  • Use clear, descriptive titles for each section
  • Keep content concise within each panel
  • Use accordions for FAQ sections and product details
  • Indicate expanded state with icon rotation
  • Allow keyboard navigation (Enter/Space to toggle)
  • Consider allowing multiple panels open simultaneously

❌ Don't

  • Don't nest accordions more than one level deep
  • Don't hide critical information in closed panels
  • Don't use accordions for navigation menus
  • Don't make panels too tall - keep content scannable
  • Don't use accordions when all content should be visible
  • Don't forget smooth transitions for better UX

Accessibility

✓

Keyboard Navigation: Enter or Space to toggle panels, Tab to move between items

✓

ARIA Attributes: Use aria-expanded and aria-controls for screen readers

✓

Focus Indicators: Clear visual focus states on accordion headers

✓

Semantic HTML: Use button elements for interactive headers
