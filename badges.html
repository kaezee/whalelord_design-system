const StatusBadge = ({ variant, icon, children }) => {
  const variants = {
    popular: 'bg-red-50 text-red-800',
    new: 'bg-sky-100 text-sky-800',
    limited: 'bg-amber-50 text-amber-800',
    seasonal: 'bg-lime-100 text-lime-800',
  };

  return (
    <span className={`inline-flex items-center gap-1 ${variants[variant]} text-xs font-semibold px-3 py-1 rounded uppercase tracking-wider`}>
      {icon && (
        <span className="inline-flex items-center justify-center w-3.5 h-3.5 text-[10px] bg-black/10 rounded-sm border border-current">
          {icon}
        </span>
      )}
      {children}
    </span>
  );
};

Used in messaging and order tracking interfaces to show current state.

React + Tailwind 
const OrderStatusBadge = ({ status, children }) => {
  const statusStyles = {
    info: 'bg-sky-100 text-sky-800',
    warning: 'bg-amber-50 text-amber-800',
    success: 'bg-emerald-100 text-emerald-800',
  };

  return (
    <span className={`inline-flex items-center ${statusStyles[status]} text-xs font-medium px-3 py-1.5 rounded-md`}>
      {children}
    </span>
  );
};

// Usage
<OrderStatusBadge status="info">Quote Received</OrderStatusBadge>
<OrderStatusBadge status="warning">Quote Pending</OrderStatusBadge>
<OrderStatusBadge status="success">Order Confirmed</OrderStatusBadge>
