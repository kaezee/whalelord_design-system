Input Fields

Text input components for forms and data entry. Includes multiple states, sizes, and validation styles.

Input Sizes

Text input fields in three sizes: small (32px), medium (40px), and large (48px).

HTML 
<!-- Small -->
<input type="text" placeholder="Enter text..." class="w-full h-8 px-3 text-sm text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10" />

<!-- Medium (Default) -->
<input type="text" placeholder="Enter text..." class="w-full h-10 px-4 text-base text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10" />

<!-- Large -->
<input type="text" placeholder="Enter text..." class="w-full h-12 px-5 text-lg text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10" />
React + Tailwind 
const Input = ({ label, size = 'md', placeholder, ...props }) => {
  const sizes = {
    sm: 'h-8 px-3 text-sm',
    md: 'h-10 px-4 text-base',
    lg: 'h-12 px-5 text-lg',
  };

  return (
    <div className="mb-5">
      {label && (
        <label className="block text-sm font-medium text-pearl-700 mb-2">
          {label}
        </label>
      )}
      <input
        type="text"
        placeholder={placeholder}
        className={`w-full ${sizes[size]} text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10 transition-all`}
        {...props}
      />
    </div>
  );
};

// Usage
<Input label="Email Address" placeholder="[email protected]" />
<Input label="Company Name" size="lg" placeholder="Enter company name..." />

React + Tailwind 
const Input = ({ label, error, success, helperText, ...props }) => {
  const getStateClasses = () => {
    if (error) return 'border-crimson-500 focus:border-crimson-500 focus:ring-crimson-500/10';
    if (success) return 'border-coral-500 focus:border-coral-500 focus:ring-coral-500/10';
    return 'border-pearl-300 focus:border-kelp-500 focus:ring-kelp-500/10';
  };

  const getHelperColor = () => {
    if (error) return 'text-crimson-500';
    if (success) return 'text-coral-500';
    return 'text-pearl-600';
  };

  return (
    <div className="mb-5">
      {label && (
        <label className="block text-sm font-medium text-pearl-700 mb-2">
          {label}
        </label>
      )}
      <input
        className={`w-full h-10 px-4 text-base text-pearl-900 bg-white border rounded-md focus:outline-none focus:ring-3 transition-all disabled:bg-pearl-100 disabled:text-pearl-500 disabled:cursor-not-allowed ${getStateClasses()}`}
        {...props}
      />
      {helperText && (
        <span className={`block text-xs mt-2 ${getHelperColor()}`}>
          {helperText}
        </span>
      )}
    </div>
  );
};

Textarea

Multi-line text input for longer content.

Maximum 500 characters
HTML 
<textarea 
  rows="4"
  placeholder="Enter your message..."
  class="w-full px-4 py-3 text-base text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10 transition-all resize-vertical"
></textarea>
React + Tailwind 
const Textarea = ({ label, helperText, rows = 4, ...props }) => (
  <div className="mb-5">
    {label && (
      <label className="block text-sm font-medium text-pearl-700 mb-2">
        {label}
      </label>
    )}
    <textarea
      rows={rows}
      className="w-full px-4 py-3 text-base text-pearl-900 bg-white border border-pearl-300 rounded-md focus:outline-none focus:border-kelp-500 focus:ring-3 focus:ring-kelp-500/10 transition-all resize-vertical"
      {...props}
    />
    {helperText && (
      <span className="block text-xs text-pearl-600 mt-2">
        {helperText}
      </span>
    )}
  </div>
);

// Usage
<Textarea 
  label="Message" 
  placeholder="Enter your message..."
  helperText="Maximum 500 characters"
  rows={4}
/>
