Snackbar / Toast

Brief notification messages that appear temporarily to provide feedback on user actions.

Snackbar Variants

Different variants for various message types: success, error, warning, info, and default.

Quote request sent successfully
Failed to process payment
Stock running low for this product
New suppliers added to marketplace
Your changes have been saved
HTML + CSS 
<!-- Success Snackbar -->
<div class="snackbar snackbar-success">
  <svg class="snackbar-icon">...</svg>
  <span>Quote request sent successfully</span>
  <button class="snackbar-close">✕</button>
</div>

<!-- Error Snackbar -->
<div class="snackbar snackbar-error">
  <svg class="snackbar-icon">...</svg>
  <span>Failed to process payment</span>
  <button class="snackbar-close">✕</button>
</div>
JSX 
import { CheckCircle, X, AlertTriangle, Info } from 'lucide-react';

const Snackbar = ({ variant = 'default', message, onClose }) => {
  const variants = {
    success: { bg: 'bg-success-500', Icon: CheckCircle },
    error: { bg: 'bg-error-500', Icon: X },
    warning: { bg: 'bg-warning-500', Icon: AlertTriangle },
    info: { bg: 'bg-waves-500', Icon: Info },
    default: { bg: 'bg-pearl-900', Icon: null }
  };

  const { bg, Icon } = variants[variant];

  return (
    <div className={`
      min-w-[320px] max-w-[600px] px-5 py-4 rounded-lg shadow-lg
      flex items-center gap-3 text-white ${bg}
    `}>
      {Icon && <Icon className="w-5 h-5 flex-shrink-0" />}
      <span className="text-sm flex-1">{message}</span>
      <button 
        onClick={onClose}
        className="opacity-80 hover:opacity-100 transition-opacity"
      >
        ✕
      </button>
    </div>
  );
};

// Usage
<Snackbar 
  variant="success" 
  message="Quote request sent successfully"
  onClose={() => console.log('Close')}
/>
