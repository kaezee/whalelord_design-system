HTML + CSS 
<!-- Horizontal Stepper -->
<div class="stepper">
  <!-- Completed Step -->
  <div class="step step-completed">
    <div class="step-circle">✓</div>
    <span class="step-label">Product Details</span>
  </div>
  
  <div class="step-line step-line-completed"></div>
  
  <!-- Active Step -->
  <div class="step step-active">
    <div class="step-circle">2</div>
    <span class="step-label">Shipping Info</span>
  </div>
  
  <div class="step-line"></div>
  
  <!-- Upcoming Steps -->
  <div class="step">
    <div class="step-circle">3</div>
    <span class="step-label">Payment</span>
  </div>
</div>
JSX 
const Stepper = ({ steps, currentStep }) => {
  return (
    <div className="flex items-center justify-between">
      {steps.map((step, index) => {
        const stepNumber = index + 1;
        const isCompleted = stepNumber < currentStep;
        const isActive = stepNumber === currentStep;
        
        return (
          <React.Fragment key={index}>
            <div className="flex flex-col items-center flex-1">
              <div className={`
                w-10 h-10 rounded-full flex items-center justify-center font-semibold
                ${isCompleted || isActive 
                  ? 'bg-success-500 border-success-500 text-white' 
                  : 'bg-white border-pearl-300 text-pearl-600'
                } border-2
              `}>
                {isCompleted ? '✓' : stepNumber}
              </div>
              <span className={`text-sm mt-2 ${
                isActive ? 'text-success-700 font-semibold' : 'text-pearl-600'
              }`}>
                {step.label}
              </span>
            </div>
            
            {index < steps.length - 1 && (
              <div className={`flex-1 h-0.5 ${
                isCompleted ? 'bg-success-500' : 'bg-pearl-300'
              }`} />
            )}
          </React.Fragment>
        );
      })}
    </div>
  );
};

// Usage
const checkoutSteps = [
  { label: 'Product Details' },
  { label: 'Shipping Info' },
  { label: 'Payment' },
  { label: 'Confirmation' }
];

<Stepper steps={checkoutSteps} currentStep={2} />
