const Stepper = ({ steps, currentStep }) => {
  return (
    <div className="flex items-center justify-between mb-10">
      {steps.map((step, index) => {
        const stepNumber = index + 1;
        const isCompleted = stepNumber < currentStep;
        const isActive = stepNumber === currentStep;
        const isUpcoming = stepNumber > currentStep;
        const isLast = index === steps.length - 1;

        return (
          <div key={index} className="flex flex-col items-center flex-1 relative">
            {/* Circle */}
            <div
              className={`w-10 h-10 rounded-full border-2 flex items-center justify-center font-semibold mb-2 z-10 ${
                isCompleted || isActive
                  ? 'bg-kelp-500 border-kelp-500 text-white'
                  : 'bg-white border-pearl-300 text-pearl-600'
              }`}
            >
              {isCompleted ? '✓' : stepNumber}
            </div>

            {/* Label */}
            <span
              className={`text-sm text-center ${
                isActive
                  ? 'text-kelp-500 font-semibold'
                  : isCompleted
                  ? 'text-pearl-700'
                  : 'text-pearl-600'
              }`}
            >
              {step.label}
            </span>

            {/* Connector Line */}
            {!isLast && (
              <div
                className={`absolute top-5 left-1/2 w-full h-0.5 -z-10 ${
                  isCompleted ? 'bg-kelp-500' : 'bg-pearl-200'
                }`}
              />
            )}
          </div>
        );
      })}
    </div>
  );
};

// Usage
const checkoutSteps = [
  { label: 'Product Details' },
  { label: 'Shipping Info' },
  { label: 'Payment' },
  { label: 'Confirmation' },
];

<Stepper steps={checkoutSteps} currentStep={2} />

const VerticalStepper = ({ steps, currentStep }) => {
  return (
    <div className="space-y-0">
      {steps.map((step, index) => {
        const stepNumber = index + 1;
        const isCompleted = stepNumber < currentStep;
        const isActive = stepNumber === currentStep;
        const isLast = index === steps.length - 1;

        return (
          <div key={index} className={`flex gap-4 ${!isLast ? 'pb-8' : ''} relative`}>
            {/* Circle and Line */}
            <div className="flex flex-col items-center">
              <div
                className={`w-10 h-10 rounded-full border-2 flex items-center justify-center font-semibold z-10 ${
                  isCompleted || isActive
                    ? 'bg-kelp-500 border-kelp-500 text-white'
                    : 'bg-white border-pearl-300 text-pearl-600'
                }`}
              >
                {isCompleted ? '✓' : stepNumber}
              </div>
              {!isLast && (
                <div
                  className={`absolute top-10 left-5 w-0.5 h-full ${
                    isCompleted ? 'bg-kelp-500' : 'bg-pearl-200'
                  }`}
                />
              )}
            </div>

            {/* Content */}
            <div className="flex-1 pt-2">
              <h4
                className={`text-base font-semibold mb-1 ${
                  isActive ? 'text-kelp-500' : isCompleted ? 'text-pearl-900' : 'text-pearl-600'
                }`}
              >
                {step.title}
              </h4>
              <p className={`text-sm ${isCompleted || isActive ? 'text-pearl-600' : 'text-pearl-500'}`}>
                {step.description}
              </p>
            </div>
          </div>
        );
      })}
    </div>
  );
};

Usage Guidelines

✅ Do

  • Use steppers for linear, sequential processes
  • Keep step labels short and action-oriented
  • Show completion status with checkmarks
  • Clearly indicate the current active step
  • Allow navigation to previous completed steps
  • Use vertical steppers on mobile or narrow layouts

❌ Don't

  • Don't use steppers for non-linear workflows
  • Don't allow skipping ahead to incomplete steps
  • Don't use more than 5-6 steps (break into sections)
  • Don't hide progress - always show all steps
  • Don't use steppers for single-page forms
  • Don't make step indicators clickable if navigation is disabled

Accessibility

✓

ARIA Labels: Use aria-label to describe progress (e.g., "Step 2 of 4")

✓

Current Status: Mark active step with aria-current="step"

✓

Visual Indicators: Don't rely solely on color to show status

✓

Keyboard Navigation: If clickable, ensure Tab and Enter key support
